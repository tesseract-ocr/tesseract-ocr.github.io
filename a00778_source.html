65 DIR128::DIR128( //from fcoord
66  const FCOORD fc //vector to quantize
67  ) {
68  int high, low, current; //binary search
69 
70  low = 0;
71  if (fc.y () == 0) {
72  if (fc.x () >= 0)
73  dir = 0;
74  else
75  dir = MODULUS / 2;
76  return;
77  }
78  high = MODULUS;
79  do {
80  current = (high + low) / 2;
81  if (dirtab[current] * fc >= 0)
82  low = current;
83  else
84  high = current;
85  }
86  while (high - low > 1);
87  dir = low;
88 }
