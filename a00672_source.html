110 // Helper function to check for a target word and handle it appropriately.
111 // Inspired by Jetsoft's requirement to process only single words on pass2
112 // and beyond.
113 // If word_config is not null:
114 // If the word_box and target_word_box overlap, read the word_config file
115 // else reset to previous config data.
116 // return true.
117 // else
118 // If the word_box and target_word_box overlap or pass <= 1, return true.
119 // Note that this function uses a fixed temporary file for storing the previous
120 // configs, so it is neither thread-safe, nor process-safe, but the assumption
121 // is that it will only be used for one debug window at a time.
122 //
123 // Since this function is used for debugging (and not to change OCR results)
124 // set only debug params from the word config file.
125 bool Tesseract::ProcessTargetWord(const TBOX& word_box,
126  const TBOX& target_word_box,
127  const char* word_config,
128  int pass) {
129  if (word_config != NULL) {
130  if (word_box.major_overlap(target_word_box)) {
131  if (backup_config_file_ == NULL) {
132  backup_config_file_ = kBackUpConfigFile;
133  FILE* config_fp = fopen(backup_config_file_, "wb");
134  ParamUtils::PrintParams(config_fp, params());
135  fclose(config_fp);
136  ParamUtils::ReadParamsFile(word_config,
137  SET_PARAM_CONSTRAINT_DEBUG_ONLY,
138  params());
139  }
140  } else {
141  if (backup_config_file_ != NULL) {
142  ParamUtils::ReadParamsFile(backup_config_file_,
143  SET_PARAM_CONSTRAINT_DEBUG_ONLY,
144  params());
145  backup_config_file_ = NULL;
146  }
147  }
148  } else if (pass > 1 && !word_box.major_overlap(target_word_box)) {
149  return false;
150  }
151  return true;

213 bool Tesseract::RecogAllWordsPassN(int pass_n, ETEXT_DESC* monitor,
214  PAGE_RES_IT* pr_it,
215  GenericVector<WordData>* words) {
216  // TODO(rays) Before this loop can be parallelized (it would yield a massive
217  // speed-up) all remaining member globals need to be converted to local/heap
218  // (eg set_pass1 and set_pass2) and an intermediate adaption pass needs to be
219  // added. The results will be significantly different with adaption on, and
220  // deterioration will need investigation.
221  pr_it->restart_page();
222  for (int w = 0; w < words->size(); ++w) {
223  WordData* word = &(*words)[w];
224  if (w > 0) word->prev_word = &(*words)[w - 1];
225  if (monitor != NULL) {
226  monitor->ocr_alive = TRUE;
227  if (pass_n == 1)
228  monitor->progress = 30 + 50 * w / words->size();
229  else
230  monitor->progress = 80 + 10 * w / words->size();
231  if (monitor->deadline_exceeded() ||
232  (monitor->cancel != NULL && (*monitor->cancel)(monitor->cancel_this,
233  words->size()))) {
234  // Timeout. Fake out the rest of the words.
235  for (; w < words->size(); ++w) {
236  (*words)[w].word->SetupFake(unicharset);
237  }
238  return false;
239  }
240  }
241  if (word->word->tess_failed) {
242  int s;
243  for (s = 0; s < word->lang_words.size() &&
244  word->lang_words[s]->tess_failed; ++s) {}
245  // If all are failed, skip it. Image words are skipped by this test.
246  if (s > word->lang_words.size()) continue;
247  }
248  // Sync pr_it with the wth WordData.
249  while (pr_it->word() != NULL && pr_it->word() != word->word)
250  pr_it->forward();
251  ASSERT_HOST(pr_it->word() != NULL);
252  WordRecognizer recognizer = pass_n == 1 ? &Tesseract::classify_word_pass1
253  : &Tesseract::classify_word_pass2;
254  classify_word_and_language(recognizer, pr_it, word);
255  if (tessedit_dump_choices) {
256  tprintf("Pass%d: %s [%s]\n", pass_n,
257  word->word->best_choice->unichar_string().string(),
258  word->word->best_choice->debug_string().string());
259  }
260  pr_it->forward();
261  }
262  return true;
263 }

289  const char* word_config,

292 
293  if (tessedit_minimal_rej_pass1) {
294  tessedit_test_adaption.set_value (TRUE);
295  tessedit_minimal_rejection.set_value (TRUE);
296  }
297 
298  if (dopasses==0 || dopasses==1) {
299  page_res_it.restart_page();
300  // ****************** Pass 1 *******************
301 
302  // Clear adaptive classifier at the beginning of the page if it is full.
303  // This is done only at the beginning of the page to ensure that the
304  // classifier is not reset at an arbitrary point while processing the page,
305  // which would cripple Passes 2+ if the reset happens towards the end of
306  // Pass 1 on a page with very difficult text.
307  // TODO(daria): preemptively clear the classifier if it is almost full.
308  if (AdaptiveClassifierIsFull()) ResetAdaptiveClassifierInternal();

311  if (sub_langs_[i]->AdaptiveClassifierIsFull())
312  sub_langs_[i]->ResetAdaptiveClassifierInternal();
313  }
314  // Set up all words ready for recognition, so that if parallelism is on
315  // all the input and output classes are ready to run the classifier.
316  GenericVector<WordData> words;
317  SetupAllWordsPassN(1, target_word_box, word_config, page_res, &words);
318  if (tessedit_parallelize) {
319  PrerecAllWordsPar(words);

322  stats_.word_count = words.size();
323 
324  stats_.dict_words = 0;
325  stats_.doc_blob_quality = 0;
326  stats_.doc_outline_errs = 0;
327  stats_.doc_char_quality = 0;
328  stats_.good_char_count = 0;
329  stats_.doc_good_char_quality = 0;
330 
331  most_recently_used_ = this;
332  // Run pass 1 word recognition.
333  if (!RecogAllWordsPassN(1, monitor, &page_res_it, &words)) return false;

335  for (page_res_it.restart_page(); page_res_it.word() != NULL;

338  fix_rep_char(&page_res_it);
339  continue;
340  }
341 
342  // Count dict words.
343  if (page_res_it.word()->best_choice->permuter() == USER_DAWG_PERM)
344  ++(stats_.dict_words);
345 
346  // Update misadaption log (we only need to do it on pass 1, since
347  // adaption only happens on this pass).
348  if (page_res_it.word()->blamer_bundle != NULL &&
349  page_res_it.word()->blamer_bundle->misadaption_debug().length() > 0) {
350  page_res->misadaption_log.push_back(
351  page_res_it.word()->blamer_bundle->misadaption_debug());
352  }
353  }
354  }
355 
356  if (dopasses == 1) return true;
357 
358  // ****************** Pass 2 *******************
359  if (tessedit_tess_adaption_mode != 0x0 && !tessedit_test_adaption &&
360  tessedit_ocr_engine_mode != OEM_CUBE_ONLY ) {
361  page_res_it.restart_page();
362  GenericVector<WordData> words;
363  SetupAllWordsPassN(2, target_word_box, word_config, page_res, &words);
364  if (tessedit_parallelize) {
365  PrerecAllWordsPar(words);
366  }

368  // Run pass 2 word recognition.
369  if (!RecogAllWordsPassN(2, monitor, &page_res_it, &words)) return false;

371

373  // doesn't set all the necessary outputs in WERD_RES.
374  if (tessedit_ocr_engine_mode == OEM_TESSERACT_ONLY ||
375  tessedit_ocr_engine_mode == OEM_TESSERACT_CUBE_COMBINED) {
376  // ****************** Pass 3 *******************
377  // Fix fuzzy spaces.
378  set_global_loc_code(LOC_FUZZY_SPACE);
379 
380  if (!tessedit_test_adaption && tessedit_fix_fuzzy_spaces
381  && !tessedit_word_for_word && !right_to_left())
382  fix_fuzzy_spaces(monitor, stats_.word_count, page_res);
383 
384  // ****************** Pass 4 *******************
385  if (tessedit_enable_dict_correction) dictionary_correction_pass(page_res);
386  if (tessedit_enable_bigram_correction) bigram_correction_pass(page_res);
387 
388  // ****************** Pass 5,6 *******************
389  rejection_passes(page_res, monitor, target_word_box, word_config);
390 
391  // ****************** Pass 7 *******************
392  // Cube combiner.
393  // If cube is loaded and its combiner is present, run it.
394  if (tessedit_ocr_engine_mode == OEM_TESSERACT_CUBE_COMBINED) {
395  run_cube_combiner(page_res);
396  }
397 
398  // ****************** Pass 8 *******************
399  font_recognition_pass(page_res);
400 
401  // ****************** Pass 9 *******************
402  // Check the correctness of the final results.
403  blamer_pass(page_res);
404  script_pos_pass(page_res);
405  }
406 
407  // Write results pass.
408  set_global_loc_code(LOC_WRITE_RESULTS);
409  // This is now redundant, but retained commented so show how to obtain
410  // bounding boxes and style information.
411 
412  // changed by jetsoft
413  // needed for dll to output memory structure
414  if ((dopasses == 0 || dopasses == 2) && (monitor || tessedit_write_unlv))
415  output_pass(page_res_it, target_word_box);
416  // end jetsoft
417  PageSegMode pageseg_mode = static_cast<PageSegMode>(
418  static_cast<int>(tessedit_pageseg_mode));
419  textord_.CleanupSingleRowResult(pageseg_mode, page_res);
420 
421  // Remove empty words, as these mess up the result iterators.
422  for (page_res_it.restart_page(); page_res_it.word() != NULL;
423  page_res_it.forward()) {
424  WERD_RES* word = page_res_it.word();
425  if (word->best_choice == NULL || word->best_choice->length() == 0)
426  page_res_it.DeleteCurrentWord();
427  }
428 
429  if (monitor != NULL) {
430  monitor->progress = 100;
431  }
432  return true;

435 void Tesseract::bigram_correction_pass(PAGE_RES *page_res) {
436  PAGE_RES_IT word_it(page_res);
437 
438  WERD_RES *w_prev = NULL;
439  WERD_RES *w = word_it.word();
440  while (1) {
441  w_prev = w;
442  while (word_it.forward() != NULL &&
443  (!word_it.word() || word_it.word()->part_of_combo)) {
444  // advance word_it, skipping over parts of combos
445  }
446  if (!word_it.word()) break;
447  w = word_it.word();
448  if (!w || !w_prev || w->uch_set != w_prev->uch_set) {
449  continue;
450  }
451  if (w_prev->word->flag(W_REP_CHAR) || w->word->flag(W_REP_CHAR)) {
452  if (tessedit_bigram_debug) {
453  tprintf("Skipping because one of the words is W_REP_CHAR\n");
454  }
455  continue;
456  }
457  // Two words sharing the same language model, excellent!
458  GenericVector<WERD_CHOICE *> overrides_word1;
459  GenericVector<WERD_CHOICE *> overrides_word2;
460 
461  STRING orig_w1_str = w_prev->best_choice->unichar_string();
462  STRING orig_w2_str = w->best_choice->unichar_string();
463  WERD_CHOICE prev_best(w->uch_set);
464  {
465  int w1start, w1end;
466  w_prev->best_choice->GetNonSuperscriptSpan(&w1start, &w1end);
467  prev_best = w_prev->best_choice->shallow_copy(w1start, w1end);
468  }
469  WERD_CHOICE this_best(w->uch_set);
470  {
471  int w2start, w2end;
472  w->best_choice->GetNonSuperscriptSpan(&w2start, &w2end);
473  this_best = w->best_choice->shallow_copy(w2start, w2end);
474  }
475 
476  if (w->tesseract->getDict().valid_bigram(prev_best, this_best)) {
477  if (tessedit_bigram_debug) {
478  tprintf("Top choice \"%s %s\" verified by bigram model.\n",
479  orig_w1_str.string(), orig_w2_str.string());
480  }
481  continue;
482  }
483  if (tessedit_bigram_debug > 2) {
484  tprintf("Examining alt choices for \"%s %s\".\n",
485  orig_w1_str.string(), orig_w2_str.string());
486  }
487  if (tessedit_bigram_debug > 1) {
488  if (!w_prev->best_choices.singleton()) {
489  w_prev->PrintBestChoices();
490  }
491  if (!w->best_choices.singleton()) {
492  w->PrintBestChoices();
493  }
494  }
495  float best_rating = 0.0;
496  int best_idx = 0;
497  WERD_CHOICE_IT prev_it(&w_prev->best_choices);
498  for (prev_it.mark_cycle_pt(); !prev_it.cycled_list(); prev_it.forward()) {
499  WERD_CHOICE *p1 = prev_it.data();
500  WERD_CHOICE strip1(w->uch_set);
501  {
502  int p1start, p1end;
503  p1->GetNonSuperscriptSpan(&p1start, &p1end);
504  strip1 = p1->shallow_copy(p1start, p1end);
505  }
506  WERD_CHOICE_IT w_it(&w->best_choices);
507  for (w_it.mark_cycle_pt(); !w_it.cycled_list(); w_it.forward()) {
508  WERD_CHOICE *p2 = w_it.data();
509  WERD_CHOICE strip2(w->uch_set);
510  {
511  int p2start, p2end;
512  p2->GetNonSuperscriptSpan(&p2start, &p2end);
513  strip2 = p2->shallow_copy(p2start, p2end);
514  }
515  if (w->tesseract->getDict().valid_bigram(strip1, strip2)) {
516  overrides_word1.push_back(p1);
517  overrides_word2.push_back(p2);
518  if (overrides_word1.size() == 1 ||
519  p1->rating() + p2->rating() < best_rating) {
520  best_rating = p1->rating() + p2->rating();
521  best_idx = overrides_word1.size() - 1;
522  }
523  }
524  }
525  }
526  if (overrides_word1.size() >= 1) {
527  // Excellent, we have some bigram matches.
528  if (EqualIgnoringCaseAndTerminalPunct(*w_prev->best_choice,
529  *overrides_word1[best_idx]) &&
530  EqualIgnoringCaseAndTerminalPunct(*w->best_choice,
531  *overrides_word2[best_idx])) {
532  if (tessedit_bigram_debug > 1) {
533  tprintf("Top choice \"%s %s\" verified (sans case) by bigram "
534  "model.\n", orig_w1_str.string(), orig_w2_str.string());
535  }
536  continue;
537  }
538  STRING new_w1_str = overrides_word1[best_idx]->unichar_string();
539  STRING new_w2_str = overrides_word2[best_idx]->unichar_string();
540  if (new_w1_str != orig_w1_str) {
541  w_prev->ReplaceBestChoice(overrides_word1[best_idx]);
542  }
543  if (new_w2_str != orig_w2_str) {
544  w->ReplaceBestChoice(overrides_word2[best_idx]);
545  }
546  if (tessedit_bigram_debug > 0) {
547  STRING choices_description;
548  int num_bigram_choices
549  = overrides_word1.size() * overrides_word2.size();
550  if (num_bigram_choices == 1) {
551  choices_description = "This was the unique bigram choice.";
552  } else {
553  if (tessedit_bigram_debug > 1) {
554  STRING bigrams_list;
555  const int kMaxChoicesToPrint = 20;
556  for (int i = 0; i < overrides_word1.size() &&
557  i < kMaxChoicesToPrint; i++) {
558  if (i > 0) { bigrams_list += ", "; }
559  WERD_CHOICE *p1 = overrides_word1[i];
560  WERD_CHOICE *p2 = overrides_word2[i];
561  bigrams_list += p1->unichar_string() + " " + p2->unichar_string();
562  if (i == kMaxChoicesToPrint) {
563  bigrams_list += " ...";
564  }
565  }
566  choices_description = "There were many choices: {";
567  choices_description += bigrams_list;
568  choices_description += "}";
569  } else {
570  choices_description.add_str_int("There were ", num_bigram_choices);
571  choices_description += " compatible bigrams.";
572  }
573  }
574  tprintf("Replaced \"%s %s\" with \"%s %s\" with bigram model. %s\n",
575  orig_w1_str.string(), orig_w2_str.string(),
576  new_w1_str.string(), new_w2_str.string(),
577  choices_description.string());
578  }

742 // Factored helper considers the indexed word and updates all the pointed
743 // values.
744 static void EvaluateWord(const PointerVector<WERD_RES>& words, int index,
745  float* rating, float* certainty, bool* bad,
746  bool* valid_permuter, int* right, int* next_left) {
747  *right = -MAX_INT32;
748  *next_left = MAX_INT32;
749  if (index < words.size()) {
750  WERD_CHOICE* choice = words[index]->best_choice;
751  if (choice == NULL) {
752  *bad = true;
753  } else {
754  *rating += choice->rating();
755  *certainty = MIN(*certainty, choice->certainty());
756  if (!Dict::valid_word_permuter(choice->permuter(), false))
757  *valid_permuter = false;
758  }
759  *right = words[index]->word->bounding_box().right();
760  if (index + 1 < words.size())
761  *next_left = words[index + 1]->word->bounding_box().left();
762  } else {
763  *valid_permuter = false;
764  *bad = true;
765  }
766 }

768 // Helper chooses the best combination of words, transferring good ones from
769 // new_words to best_words. To win, a new word must have (better rating and
770 // certainty) or (better permuter status and rating within rating ratio and
771 // certainty within certainty margin) than current best.
772 // All the new_words are consumed (moved to best_words or deleted.)
773 // The return value is the number of new_words used minus the number of
774 // best_words that remain in the output.
775 static int SelectBestWords(double rating_ratio,
776  double certainty_margin,
777  bool debug,
778  PointerVector<WERD_RES>* new_words,
779  PointerVector<WERD_RES>* best_words) {
780  // Process the smallest groups of words that have an overlapping word
781  // boundary at the end.
782  GenericVector<WERD_RES*> out_words;
783  // Index into each word vector (best, new).
784  int b = 0, n = 0;
785  int num_best = 0, num_new = 0;
786  while (b < best_words->size() || n < new_words->size()) {
787  // Start of the current run in each.
788  int start_b = b, start_n = n;
789  // Rating of the current run in each.
790  float b_rating = 0.0f, n_rating = 0.0f;
791  // Certainty of the current run in each.
792  float b_certainty = 0.0f, n_certainty = 0.0f;
793  // True if any word is missing its best choice.
794  bool b_bad = false, n_bad = false;
795  // True if all words have a valid permuter.
796  bool b_valid_permuter = true, n_valid_permuter = true;
797 
798  while (b < best_words->size() || n < new_words->size()) {
799  int b_right = -MAX_INT32;
800  int next_b_left = MAX_INT32;
801  EvaluateWord(*best_words, b, &b_rating, &b_certainty, &b_bad,
802  &b_valid_permuter, &b_right, &next_b_left);
803  int n_right = -MAX_INT32;
804  int next_n_left = MAX_INT32;
805  EvaluateWord(*new_words, n, &n_rating, &n_certainty, &n_bad,
806  &n_valid_permuter, &n_right, &next_n_left);
807  if (MAX(b_right, n_right) < MIN(next_b_left, next_n_left)) {
808  // The word breaks overlap. [start_b,b] and [start_n, n] match.
809  break;
810  }
811  // Keep searching for the matching word break.
812  if ((b_right < n_right && b < best_words->size()) ||
813  n == new_words->size())
814  ++b;
815  else
816  ++n;
817  }
818  bool new_better = false;
819  if (!n_bad && (b_bad || (n_certainty > b_certainty &&
820  n_rating < b_rating) ||
821  (!b_valid_permuter && n_valid_permuter &&
822  n_rating < b_rating * rating_ratio &&
823  n_certainty > b_certainty - certainty_margin))) {
824  // New is better.
825  for (int i = start_n; i <= n; ++i) {
826  out_words.push_back((*new_words)[i]);
827  (*new_words)[i] = NULL;
828  ++num_new;
829  }
830  new_better = true;
831  } else if (!b_bad) {
832  // Current best is better.
833  for (int i = start_b; i <= b; ++i) {
834  out_words.push_back((*best_words)[i]);
835  (*best_words)[i] = NULL;
836  ++num_best;
837  }
838  }
839  int end_b = b < best_words->size() ? b + 1 : b;
840  int end_n = n < new_words->size() ? n + 1 : n;
841  if (debug) {
842  tprintf("%d new words %s than %d old words: r: %g v %g c: %g v %g"
843  " valid dict: %d v %d\n",
844  end_n - start_n, new_better ? "better" : "worse",
845  end_b - start_b, n_rating, b_rating,
846  n_certainty, b_certainty, n_valid_permuter, b_valid_permuter);
847  }
848  // Move on to the next group.
849  b = end_b;
850  n = end_n;
851  }
852  // Transfer from out_words to best_words.
853  best_words->clear();
854  for (int i = 0; i < out_words.size(); ++i)
855  best_words->push_back(out_words[i]);
856  return num_new - num_best;
857 }

900 // Generic function for classifying a word. Can be used either for pass1 or
901 // pass2 according to the function passed to recognizer.
902 // word_data holds the word to be recognized, and its block and row, and
903 // pr_it points to the word as well, in case we are running LSTM and it wants
904 // to output multiple words.
905 // Recognizes in the current language, and if successful that is all.
906 // If recognition was not successful, tries all available languages until
907 // it gets a successful result or runs out of languages. Keeps the best result.
908 void Tesseract::classify_word_and_language(WordRecognizer recognizer,
909  PAGE_RES_IT* pr_it,
910  WordData* word_data) {
911  // Best result so far.
912  PointerVector<WERD_RES> best_words;
913  // Points to the best result. May be word or in lang_words.
914  WERD_RES* word = word_data->word;
915  clock_t start_t = clock();
916  if (classify_debug_level || cube_debug_level) {
917  tprintf("%s word with lang %s at:",
918  word->done ? "Already done" : "Processing",
919  most_recently_used_->lang.string());
920  word->word->bounding_box().print();
921  }
922  if (word->done) {
923  // If done on pass1, leave it as-is.
924  if (!word->tess_failed)
925  most_recently_used_ = word->tesseract;
926  return;
927  }
928  int sub = sub_langs_.size();
929  if (most_recently_used_ != this) {
930  // Get the index of the most_recently_used_.
931  for (sub = 0; sub < sub_langs_.size() &&
932  most_recently_used_ != sub_langs_[sub]; ++sub) {}
933  }
934  most_recently_used_->RetryWithLanguage(
935  *word_data, recognizer, &word_data->lang_words[sub], &best_words);
936  Tesseract* best_lang_tess = most_recently_used_;
937  if (!WordsAcceptable(best_words)) {
938  // Try all the other languages to see if they are any better.
939  if (most_recently_used_ != this &&
940  this->RetryWithLanguage(*word_data, recognizer,
941  &word_data->lang_words[sub_langs_.size()],
942  &best_words) > 0) {
943  best_lang_tess = this;
944  }
945  for (int i = 0; !WordsAcceptable(best_words) && i < sub_langs_.size();
946  ++i) {
947  if (most_recently_used_ != sub_langs_[i] &&
948  sub_langs_[i]->RetryWithLanguage(*word_data, recognizer,
949  &word_data->lang_words[i],
950  &best_words) > 0) {
951  best_lang_tess = sub_langs_[i];
952  }
953  }
954  }
955  most_recently_used_ = best_lang_tess;
956  if (!best_words.empty()) {
957  if (best_words.size() == 1 && !best_words[0]->combination) {
958  // Move the best single result to the main word.
959  word_data->word->ConsumeWordResults(best_words[0]);
960  } else {
961  // Words came from LSTM, and must be moved to the PAGE_RES properly.
962  word_data->word = best_words.back();
963  pr_it->ReplaceCurrentWord(&best_words);
964  }
965  ASSERT_HOST(word_data->word->box_word != NULL);
966  } else {
967  tprintf("no best words!!\n");
968  }
969  clock_t ocr_t = clock();
970  if (tessedit_timing_debug) {
971  tprintf("%s (ocr took %.2f sec)\n",
972  word->best_choice->unichar_string().string(),
973  static_cast<double>(ocr_t-start_t)/CLOCKS_PER_SEC);
974  }
975 }

1280

1358  char_set.get_islower(s + offset, lengths[i]) &&

1361  offset += lengths[i++];
1362  }
