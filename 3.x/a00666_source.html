126  // when the baseline wanders around. I've decided to force every
127  // word to match the (straight) baseline. The math below is just
128  // projecting the word origin onto the baseline. All numbers are

130  // the bottom left and the unit is points, which is 1/72 inch.

139  y = line_y1;
140  } else {

144  y = line_y2 + t * (line_y2 - line_y1);

156  // which has the origin in the bottom left. The Tesseract
157  // coordinate system has the origin in the upper left.
158  //
159  // PDF is kind of a like turtle graphics, and we orient the

161  // transformation.
162  //
163  // Rotate RTL Translate
164  //
165  // [ x' y' 1 ] = [ x y 1 ] [ cosùúÉ -sinùúÉ 0 ] [ -1 0 0 ] [ 1 0 0 ]
166  // [ sinùúÉ cosùúÉ 0 ] [ 0 1 0 ] [ 0 1 0 ]
167  // [ 0 0 1 ] [ 0 0 1 ] [ x y 1 ]
168  //

170  static_cast<double>(line_x2 - line_x1));

172  a = cos(theta);
173  b = sin(theta);
174  c = -sin(theta);
175  d = cos(theta);
176  switch(writing_direction) {
